<!--
  =================================================================
  APP SPEC TEMPLATE — fill this in for your new project
  =================================================================

  使用方式：
    cp app_spec.template.txt app_spec.txt
    然后填充每一段；标 [REQUIRED] 的必填，[OPTIONAL] 可整段删掉。

  这份文件会被 Initializer Agent 读取一次，并据此生成 ~80 条 e2e
  测试用例（feature_list.json）。**写得越具体，生成的测试越准。**

  几条经验：
  1. 钉死技术栈与端口。不要让 Agent 在 React / Vue 之间自己选；
     不要让它在 3000/5173/8080 之间随机挑——后续截图脚本要写死 URL。
  2. 钉死颜色（hex）和字体。"现代风格" 是不可验证的；
     "#4f46e5 indigo, Inter font" 是可验证的。
  3. <features> 段每条 feature 列出具体行为、边界条件、错误处理。
     "用户能登录" 太虚；"邮箱+密码表单，邮箱格式校验，成功跳 /dashboard，
     失败在表单下方显示红色错误" 才能拆成测试。
  4. 不要自己写测试用例——Initializer 会从 spec 推。你的活是写 spec。
  5. 当前 harness 用 Puppeteer 截图作为证据，所以这套模板默认目标是
     **Web 应用**。其他形态（CLI / 桌面）需要额外改 hooks。
  =================================================================
-->

  <overview>
    {{ 一段话描述应用的高层目标，例如：
       "一个面向开发者的本地代码笔记应用，支持 Markdown + 代码高亮，
        数据全部本地存储，不联网。" }}
  </overview>

  <!-- [REQUIRED] 钉死技术栈，避免 Agent 临场选型导致每轮风格漂移。 -->
  <technology_stack>

      <framework>{{ 例：React + Vite / Next.js / Vue 3 + Vite / 原生 HTML }}</framework>
      <styling>{{ 例：Tailwind CSS (CDN) / CSS Modules / styled-components }}</styling>
      <state_management>{{ 例：React hooks / Redux Toolkit / Zustand / none }}</state_management>
      <routing>{{ 例：React Router v6 / Next.js File Routing / none }}</routing>
      <port>{{ 唯一启动端口，例 5173 }}</port>

    <backend>{{ 例："None — 纯客户端" / "Express + SQLite，端口 3001" }}</backend>
    <data_storage>{{ 例：localStorage / IndexedDB / SQLite / Postgres / none }}</data_storage>
  </technology_stack>

         - 多页站点：<pages><page id="...">

  <features>

        - {{ 错误处理，例："JSON 非法时高亮错误行号并展示原因" }}
        - {{ UI 反馈，例："复制成功时弹 'Copied!' toast，2 秒后消失" }}
        - {{ 可访问性，例："所有按钮支持键盘 Tab 聚焦" }}

        - {{ ... }}

  </features>

  <ui_layout>
    <main_structure>
      {{ 例：
         - 两栏布局：左侧 240px sidebar + 右侧主内容区

         - ...（如不支持暗色模式可删除）}}

    <typography>
      {{ 例：
         - 正文：Inter, system-ui, sans-serif

    </typography>

    <components>
      {{ 描述按钮、表单、输入框等通用组件的规范，

    </components>
  </design_system>

  <!-- [REQUIRED] Evaluator 会显式读这段。
       条目要"从外部可观察"——不要写"用了 useMemo"这种内部细节。 -->
  <success_criteria>
    <functionality>

      - {{ 例：表单验证、错误提示、空状态都有处理 }}
      - {{ 例：无 console error / warning }}
    </functionality>

    <user_experience>
      - {{ 例：响应式适配移动端 }}
      - {{ 例：交互延迟 < 100ms，无加载转圈 }}
      - {{ 例：键盘可达，支持 Tab / Enter / Escape }}
    </user_experience>

    <technical_quality>
      - {{ 例：每个 feature 一个独立组件 }}

    </technical_quality>
  </success_criteria>

  <privacy>

       - /privacy 页面：列出数据收集说明、第三方服务、联系方式

</project_specification>
